async function chatWithClaudeHaiku() {
  try {
    const response = await client.chat.completions.create({
      model: 'claude-haiku',  // 模型名直接填Claude Haiku
      messages: [
        { role: 'user', content: '请用中文写一段关于人工智能未来发展的简短思考。' }
      ],
      max_tokens: 500,
      temperature: 0.7,
    });

    console.log('Claude Haiku回复：', response.choices[0].message.content);
  } catch (error) {
    console.error('调用失败：', error.message);

java
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;

public class YunwuDemo {
    public static void main(String[] args) throws Exception {
        // 1. 创建HTTP客户端
        HttpClient client = HttpClient.newHttpClient();
        
        // 2. 构造请求体（JSON）
        String json = """
            {
                "model": "o4-mini",
                "messages": [
                    {"role": "system", "content": "你是一个Java助手"},
                    {"role": "user", "content": "用Java写一个递归计算斐波那契数列的例子"}
                ],
                "temperature": 0.7
            }
            """;
        
        // 3. 构建POST请求
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://www.yunwuai.cc/v1/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer YOUR_YUNWU_API_KEY") // 替换成你的key
                .POST(BodyPublishers.ofString(json))
                .build();
        
        // 4. 发送请求并获取响应
        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
        
        // 5. 打印结果
        System.out.println("状态码: " + response.statusCode());
        System.out.println("返回内容: " + response.body());
    }
}

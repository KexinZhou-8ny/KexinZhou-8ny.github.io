def call_qwen_turbo(prompt):
    try:
        response = client.chat.completions.create(
            model="qwen-turbo",  # 这里写云雾平台支持的通义千问模型名
            messages=[
                {"role": "system", "content": "你是一个有用的助手，可以回答任何问题。"},
                {"role": "user", "content": prompt}
            ],
            temperature=0.7,
            max_tokens=500
        )
        return response.choices[0].message.content
    except Exception as e:
        return f"请求出错了：{e}"

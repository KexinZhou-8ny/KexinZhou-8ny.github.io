## 接入 Java 示例：改一行代码就够了

[云雾ai聚合站](https://www.yunwuai.cc/)完全兼容 OpenAI 格式，Java 开发者只需使用 `HttpURLConnection` 或 `OkHttp` 即可。下面给一个完整示例（使用 JDK 11+ 的 `HttpClient`）：

java
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;

public class GPT41JavaExample {
    public static void main(String[] args) throws Exception {
        // 只需替换 base_url 和 apiKey
        String apiKey = "你的云雾API Key";
        String baseUrl = "https://www.yunwuai.cc/v1"; // [云雾ai聚合站](https://www.yunwuai.cc/)提供的端点

        HttpClient client = HttpClient.newHttpClient();

        String requestBody = """
            {
                "model": "gpt-4.1", // 或 gpt-4.1-mini
                "messages": [
                    {"role": "user", "content": "请用Java写一个单例模式"}
                ],
                "max_tokens": 1024,
                "temperature": 0.7
            }
            """;

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(baseUrl + "/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + apiKey)
                .POST(BodyPublishers.ofString(requestBody))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());

        System.out.println("返回结果：");
        System.out.println(response.body());
    }
}

### 第四步：处理企业级需求——并发与超时

企业接入场景下，通常需要控制并发频率和超时时间。下面这段代码展示了如何在Node.js里优雅地处理这些情况。

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  apiKey: 'sk-你的云雾API中转站Key',
  baseURL: 'https://www.yunwuai.cc/v1',
  timeout: 30000, // 30秒超时
  maxRetries: 2,  // 最多重试2次
});

async function batchCall(prompts) {
  const results = [];
  const concurrencyLimit = 5; // 控制并发数为5
  const queue = [...prompts];

  async function worker() {
    while (queue.length > 0) {
      const prompt = queue.shift();
      try {
        const completion = await client.chat.completions.create({
          model: 'gpt-5.2',
          messages: [
            { role: 'user', content: prompt }
          ],
          max_tokens: 200,
        });
        results.push(completion.choices[0].message.content);
      } catch (err) {
        console.error(`处理prompt "${prompt.substring(0, 20)}..." 时出错:`, err.message);
        results.push(null);
      }
    }
  }

  const workers = Array(concurrencyLimit).fill(null).map(() => worker());
  await Promise.all(workers);
  return results;
}

// 使用示例
const inputPrompts = [
  '写一段Node.js的错误处理代码',
  '解释一下Promise的工作原理',
  '给我一个JSON to Schema的转换函数',
  // ... 可以继续加prompt
];

batchCall(inputPrompts).then((responses) => {
  responses.forEach((resp, idx) => {
    if (resp) {
      console.log(`\n--- 回答 ${idx + 1} ---\n${resp.substring(0, 200)}...`);
    }
  });
});

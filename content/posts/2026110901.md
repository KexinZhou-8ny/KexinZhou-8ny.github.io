云雾API聚合平台的兼容性做得非常极致。只要是兼容 OpenAI SDK 的 Java 库，比如 `openai-java`，改个 `baseUrl` 就完事。

下面是完整的 **Java 接入 GPT-5.1 接口示例**：

java
import com.theokanning.openai.service.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.util.List;

public class YunWuGPT51Demo {
    public static void main(String[] args) {
        // 👇 只改这一行：把官方地址换成云雾的地址
        String baseUrl = "https://www.yunwuai.cc/v1";
        // 你的云雾 API Key（注册后获得）
        String apiKey = "sk-你的云雾API密钥";

        // 创建 OpenAiService 实例
        OpenAiService service = new OpenAiService(apiKey, baseUrl, Duration.ofSeconds(30));

        // 构建请求，使用 GPT-5.1 模型
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-5.1") // 云雾平台支持的模型名称
                .messages(List.of(
                        new ChatMessage("system", "你是一个专业的Java技术顾问。"),
                        new ChatMessage("user", "用Java实现一个并发安全的计数器，要求线程安全。")
                ))
                .maxTokens(2048)
                .temperature(0.7)
                .build();

        // 发送请求并获取回复
        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> {
                    System.out.println(choice.getMessage().getContent());
                });
    }
}


运行这段代码，根本不需要翻墙，直接就能在国内服务器上拿到 GPT-5.1 的回复。

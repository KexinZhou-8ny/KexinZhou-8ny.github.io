### 核心代码实现

下面这段代码，就是调用Grok 3 Mini API的完整Java示例。我把每一行都加了注释，确保你能看懂。

java
import com.openai.client.OpenAIClient;
import com.openai.client.okhttp.OpenAIOkHttpClient;
import com.openai.models.chat.completions.ChatCompletion;
import com.openai.models.chat.completions.ChatCompletionCreateParams;
import com.openai.models.chat.completions.ChatMessage;
import com.openai.models.chat.completions.ChatMessageUser;

import java.util.Arrays;

public class Grok3MiniExample {

    public static void main(String[] args) {
        // 1. 从环境变量读取 API Key，安全第一
        String apiKey = System.getenv("YUNWU_API_KEY");
        if (apiKey == null || apiKey.isEmpty()) {
            System.out.println("请先设置环境变量 YUNWU_API_KEY，值为你在云雾平台获取的 API Key");
            return;
        }

        // 2. 创建 OpenAI 客户端，并指定 baseUrl 为云雾平台的地址
        OpenAIClient client = OpenAIOkHttpClient.builder()
                .apiKey(apiKey)
                .baseUrl("https://www.yunwuai.cc/v1") // 关键！这里换成云雾的地址
                .build();

        // 3. 构建聊天消息
        ChatMessage userMessage = ChatMessage.builder()
                .content(ChatMessageContent.ofTextContent("请用一句话介绍你自己，并说出你今天能为人类做什么？"))
                .role(ChatMessage.Role.USER)
                .build();

        // 4. 构建 API 请求参数
        ChatCompletionCreateParams params = ChatCompletionCreateParams.builder()
                .model("grok-3-mini") // 模型名称，一定要填对
                .messages(Arrays.asList(userMessage))
                .maxTokens(300) // 限制输出长度，防止意外超额
                .temperature(0.7) // 控制创造力
                .build();

        // 5. 发送请求并获取响应
        ChatCompletion completion = client.chat().completions().create(params);

        // 6. 打印返回结果
        System.out.println("Grok 3 Mini 回答：");
        completion.choices().forEach(choice -> {
            System.out.println(choice.message().content().text());
        });
    }
}

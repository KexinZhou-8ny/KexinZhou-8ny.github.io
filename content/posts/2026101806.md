javascript
const axios = require('axios');

// 重复请求缓存：用请求内容的哈希值作为key，存储请求结果
const requestCache = new Map();
const CACHE_TTL = 60 * 1000; // 缓存有效期：1分钟

// 生成请求内容的简单哈希（用于检测重复）
function generateHash(payload) {
  return require('crypto').createHash('md5').update(JSON.stringify(payload)).digest('hex');
}

/**
 * 调用豆包API，并自动避免重复请求
 * @param {string} prompt - 提示内容
 * @param {object} options - 可选参数：model, max_tokens, temperature 等
 * @returns {Promise<string>} - 返回AI的回复内容
 */
async function callDoubaoApi(prompt, options = {}) {
  // 1. 构建请求载荷
  const payload = {
    model: options.model || 'doubao-pro', // 指定豆包模型，也可以设为doubao-turbo等
    messages: [
      { role: 'user', content: prompt }
    ],
    max_tokens: options.max_tokens || 500,
    temperature: options.temperature || 0.7,
  };

  // 2. 生成哈希，检测重复请求
  const hash = generateHash(payload);
  if (requestCache.has(hash)) {
    const cachedEntry = requestCache.get(hash);
    if (Date.now() - cachedEntry.timestamp < CACHE_TTL) {
      console.log('✅ 检测到重复请求，直接返回缓存结果，省钱了！');
      return cachedEntry.data;
    } else {
      requestCache.delete(hash); // 缓存超时，删除
    }
  }

  // 3. 发起真实的API请求
  try {
    const response = await axios({
      method: 'post',
      url: 'https://www.yunwuai.cc/v1/chat/completions', // 核心：改成[云雾api聚合站](https://www.yunwuai.cc/)的接口
      headers: {
        'Authorization': `Bearer ${process.env.YUNWU_API_KEY}`, // 环境变量中设置你的API Key
        'Content-Type': 'application/json',
      },
      data: payload,
      timeout: 30000, // 30秒超时
    });

    // 4. 解析结果
    const reply = response.data.choices[0].message.content;

    // 5. 存入缓存
    requestCache.set(hash, {
      data: reply,
      timestamp: Date.now(),
    });

    return reply;
  } catch (error) {
    console.error('❌ API请求失败:', error.message);
    // 如果遇到网络错误（如超时、连接拒绝），可以触发重试逻辑（但这里不自动重试，避免额外开销）
    throw new Error(`豆包API调用失败: ${error.message}`);
  }
}

// 使用示例
(async () => {
  // 设置API Key (建议在.env或者环境变量中设置)
  // process.env.YUNWU_API_KEY = '你的API Key';

  // 模拟第一次调用
  const result1 = await callDoubaoApi('请用50个字介绍豆包模型');
  console.log('结果1:', result1);

  // 模拟第二次相同请求（会命中缓存，不会消耗Token）
  const result2 = await callDoubaoApi('请用50个字介绍豆包模型');
  console.log('结果2:', result2); // 输出与result1相同，但没扣钱

  // 模拟不同请求（不会命中缓存）
  const result3 = await callDoubaoApi('用30个字说一个冷笑话');
  console.log('结果3:', result3);
})();

### 关键点解释

- **重复请求拦截**：代码通过生成请求内容的MD5哈希值，并存入缓存。如果30秒内发起完全相同（prompt和参数完全一致）的请求，会直接返回缓存结果，而不是再次调用API。这在用户点击多次提交按钮或前端轮询场景下，能狠狠省钱。
- **超时机制**：设置30秒超时，避免无限等待导致服务器资源浪费。
- **错误处理**：捕获网络错误并抛出，不做盲目自动重试（避免重复消耗）。

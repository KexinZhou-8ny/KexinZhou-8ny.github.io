这是重点。我假设你已经有Node.js环境（如果没有，装个Node 18+）。创建一个新项目并安装openai库：

然后写一个chat.js：

javascript
import OpenAI from 'openai';

const client = new OpenAI({
  baseURL: 'https://www.yunwuai.cc/v1', // 替换为[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)的API地址

async function main() {

    model: 'qwen-plus',
    messages: [
      { role: 'user', content: '用Node.js写一个斐波那契数列函数，要求使用记忆化递归' },
    ],
    max_tokens: 1024,
  });
  console.log(completion.choices[0].message.content);
}

main();

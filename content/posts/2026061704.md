const OpenAI = require('openai');

// 1. 配置[云雾api聚合平台](https://www.yunwuai.cc/)的客户端
const client = new OpenAI({
    // 核心：将baseURL指向[云雾api聚合平台](https://www.yunwuai.cc/)
    baseURL: "https://www.yunwuai.cc/v1",
    // 你的API密钥，注意保密
    apiKey: "YOUR_YUNWUAI_API_KEY"
});

// 2. 定义优化函数
async function optimizeCode(badCodeSnippet) {
    try {
        const response = await client.chat.completions.create({
            // 调用DeepSeek V3.2 满血版
            model: "deepseek-chat",
            messages: [
                {
                    role: "system",
                    content: "你是一位顶级的JavaScript代码优化专家。你的任务是：分析用户提供的代码，找出其中可以优化的地方（性能、可读性、安全性），然后直接输出优化后的完整代码。请用javascript标签包裹输出。"
                },
                {
                    role: "user",
                    content: `请优化以下JavaScript代码：\n\`\`\`javascript\n${badCodeSnippet}\n\`\`\``
                }
            ],
            // 设置较高温度以获得更有创造力的优化方案
            temperature: 0.5,
            // 最大Token长度，DeepSeek V3.2支持32K上下文，这很靠谱
            max_tokens: 4096
        });

        // 3. 提取并返回优化结果
        const optimizedCode = response.choices[0]?.message?.content;
        return optimizedCode;

    } catch (error) {
        console.error("调用API时出错:", error);
        return null;
    }
}

// 4. 测试：给出一段需要优化的代码
const badCode = `
function loop(arr) {
    for (let i = 0; i < arr.length; i++) {
        console.log(arr[i]);
    }
}
`;

(async () => {
    console.log("正在调用DeepSeek V3.2进行代码优化...");
    const result = await optimizeCode(badCode);
    if (result) {
        console.log("优化结果如下：");
        console.log(result);
    } else {
        console.log("优化失败，请检查API密钥或网络连接。");
    }
})();

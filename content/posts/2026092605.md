java

import java.io.*;
import java.net.HttpURLConnection;
import java.net.URL;
import java.nio.charset.StandardCharsets;

/**
 * 示例：使用[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)接口，在Java中调用Claude Sonnet
 * 核心：只需更改 BASE_URL 和 API_KEY，无需任何其他配置
 */
public class ClaudeSonnetDemo {

    // ★ 最重要的核心：修改BASE_URL为你接入点的地址
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    // ★ 你的API Key，在云雾后台直接生成
    private static final String API_KEY = "your_api_key_here";

    public static void main(String[] args) throws IOException {
        String model = "claude-3-sonnet-20240229"; // 你想调用的模型，这里也可以换别的
        String userMessage = "请用Java写一个单例模式的例子，并解释线程安全。";

        String response = callAI(model, userMessage);
        System.out.println("AI Response: " + response);
    }

    /**
     * 通用调用方法，支持所有OpenAI兼容接口
     */
    public static String callAI(String model, String userMessage) throws IOException {
        URL url = new URL(BASE_URL + "/chat/completions");
        HttpURLConnection connection = (HttpURLConnection) url.openConnection();
        connection.setRequestMethod("POST");
        connection.setRequestProperty("Content-Type", "application/json");
        connection.setRequestProperty("Authorization", "Bearer " + API_KEY);
        connection.setDoOutput(true);

        // 构建请求体 (OpenAI标准格式)
        String jsonInputString = "{\"model\": \"" + model + "\", \"messages\": [{\"role\": \"user\", \"content\": \"" + userMessage + "\"}]}";

        try (OutputStream os = connection.getOutputStream()) {
            byte[] input = jsonInputString.getBytes(StandardCharsets.UTF_8);
            os.write(input, 0, input.length);
        }

        // 读取响应
        try (BufferedReader br = new BufferedReader(new InputStreamReader(connection.getInputStream(), StandardCharsets.UTF_8))) {
            StringBuilder response = new StringBuilder();
            String responseLine;
            while ((responseLine = br.readLine()) != null) {
                response.append(responseLine.trim());
            }
            return response.toString();
        }
    }
}

**你看，就这么简单！** 代码里唯一需要你改的，只有 `BASE_URL` 和 `API_KEY`。之前你为了调不同的平台，可能得准备四五套不同的认证逻辑、不同的请求格式。现在？

**全部统一了。**

如果你想换成GPT-4o，只需把代码中的 `model` 改成 `"gpt-4o"` 即可。想换成DeepSeek-R1？更简单，模型名换成 `"deepseek-r1"`。

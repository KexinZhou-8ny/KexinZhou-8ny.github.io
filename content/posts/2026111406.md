import fetch from 'node-fetch';

async function main() {
  const apiKey = "sk-你的云雾API_KEY_HERE"; // 替换你的Key
  const baseUrl = "https://www.yunwuai.cc/v1"; // 中转站地址

  const body = {
    model: "gemini-3-pro",
    messages: [
      { role: "user", content: "你好，给我写一首关于夏天的简短诗歌。" }
    ],
    stream: false,
  };

  try {
    const response = await fetch(`${baseUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${apiKey}`,
      },
      body: JSON.stringify(body),
    });

    if (!response.ok) {
      throw new Error(`HTTP 错误! 状态码: ${response.status}, 消息: ${await response.text()}`);
    }

    const data = await response.json();
    console.log("AI 回复:", data.choices[0].message.content);
  } catch (error) {
    console.error("调用失败:", error);
  }
}

main();

## 99%的报错都在这了：排雷指南

我知道，你可能会遇到一些报错。别慌，我把最常出现的几种问题以及它们的解法都列出来了。

### 错误1：`401 Authentication Error` (认证错误)

**症状**：HTTP 401 状态码，提示认证失败。

**原因**：99.99%是因为你的 `apiKey` 填错了。要么是复制漏了字符，要么是Key过期了，要么是你直接把OpenAI的Key填进来了。

**解法**：
1.  去[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)控制台重新生成一个Key。
2.  仔细检查代码里 `apiKey` 变量的前后，有没有不小心引入了空格或换行。用引号括起来，保证它是一个干净的字符串。

### 错误2：`404 Not Found` 或 `模型不存在`

**症状**：请求发送成功，但返回模型不存在。

**原因**：`model` 参数写错了。中转站的模型ID和官方的不完全一样，你需要去[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)的文档或模型列表里查找确切的ID。

**解法**：
1.  访问[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)的文档，搜索“Gemini 3 Pro”或“模型列表”。
2.  查找类似 `gemini-3-pro` 或 `gemini-3.0-pro` 的字段。请务必以官方文档为准，我这里给出的 `gemini-3-pro` 只是示例。
3.   `baseURL` 写错了。请确保是 `https://www.yunwuai.cc/v1` 而不是其他。

### 错误3：`202 Not Modified` 或 长时间无响应

**症状**：请求发送出去后，要么直接返回202状态码，要么一直卡住，最后超时。

**原因**：常见于开启了流式(`stream: true`)输出，但你的代码或网络环境没有正确处理流式数据。也可能是因为请求的Prompt太长，模型处理需要时间。

**解法**：
1.  调试期间，请务必先把 `stream` 设置为 `false`，使用非流式输出，确认整体链路没问题。
2.  检查输出端代码。如果你用 `fetch` 方案二，记得不要直接用 `response.json()` 去解析流式数据。你需要监听 `response.body` 的 `data` 事件。

### 错误4：`400 Bad Request` (请求体格式错误)

**症状**：HTTP 400 状态码。

**原因**：你的JSON请求体格式不对。常见是 `messages` 的 `role` 只写了 `user` 和 `assistant`，没有系统的 `system` 提示词可以，但不能为空。或者你传了 `images` 但格式不对。

**解法**：
1.  检查JSON语法。用在线JSON格式化工具检查你的body字符串。
2.  确保 `messages` 至少有一个 `user` 消息，并且 `content` 是字符串，不是数组。
3.  如果你要传图片，请严格按照OpenAI视觉API的格式：`content: [{ type: "text", text: "..." }, { type: "image_url", image_url: { url: "data:image/png;base64,..." } }]`。

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;
import java.nio.charset.StandardCharsets;

public class CallClaudeViaYunwu {
    // 重点：替换成你自己在云雾充值后的秘钥
    private static final String API_KEY = "YOUR_YUNWU_API_KEY_HERE";
    // 重点：把api地址从 “https://api.openai.com/v1” 换成这款聚平台的地址
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";

    public static void main(String[] args) {
        try {
            // 1. 构造请求的JSON字符串（符合OpenAI兼容格式）
            String jsonInput = """
                {
                  "model": "claude-3-5-sonnet-20241022",
                  "messages": [
                    {"role": "system", "content": "你是一个Java编程助手。"},
                    {"role": "user", "content": "帮我写一个可以反转字符串的简单方法？"}
                  ],
                  "max_tokens": 300,
                  "temperature": 0.7
                }
                """;
            // 2. 建立HTTP连接
            URL url = new URL(BASE_URL + "/chat/completions");
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("Authorization", "Bearer " + API_KEY);
            conn.setRequestProperty("Content-Type", "application/json");
            conn.setDoOutput(true);

            // 3. 写入请求体
            try (OutputStream os = conn.getOutputStream()) {
                byte[] input = jsonInput.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            // 4. 读取响应（支持流式响应，这里先做普通的非流式读取）
            int responseCode = conn.getResponseCode();
            System.out.println("响应码: " + responseCode);
            BufferedReader br;
            if (responseCode == 200) {
                br = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8));
            } else {
                br = new BufferedReader(new InputStreamReader(conn.getErrorStream(), StandardCharsets.UTF_8));
            }
            StringBuilder response = new StringBuilder();
            String responseLine;
            while ((responseLine = br.readLine()) != null) {
                response.append(responseLine.trim());
            }
            br.close();
            conn.disconnect();
            // 5. 打印结果
            System.out.println("Claude 回复：");
            System.out.println(response.toString());

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

另外，如果你的调用量变大，遇到偶尔的超时或错误，可以在你的请求代码中加入重试逻辑，提高稳定性：

python
import time
import openai

def call_with_retry(model, messages, max_retries=3):
    for i in range(max_retries):
        try:
            client = openai.OpenAI(api_key="你的API Key", base_url="https://www.yunwuai.cc/v1")
            response = client.chat.completions.create(model=model, messages=messages)
            return response
        except Exception as e:
            print(f"第{i+1}次调用失败: {e}")
            time.sleep(2 ** i)
    raise Exception("多次调用失败")

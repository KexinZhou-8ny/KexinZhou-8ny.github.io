    @Value("${ai.base-url}")
    private String baseUrl;
    @Value("${ai.api-key}")
    private String apiKey;

    public String generateCode(String prompt) {
        // 使用OpenAI客户端，将baseUrl替换为[[云雾ai官网](https://www.yunwuai.cc/)](https://www.yunwuai.cc/)地址

                .baseUrl(baseUrl)
                .apiKey(apiKey)
                .build();

                .model("Qwen3-Coder") // 直接指定模型ID
                .messages(List.of(new SystemMessage("你是顶级Java架构师，擅长生成高质量、可维护的Java代码。"), new UserMessage(prompt)))
                .build();

        return client.chatCompletion(request)
                .getChoices().get(0).getMessage().getContent();
    }
}
